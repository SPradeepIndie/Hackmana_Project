CREATE DATABASE hakmanaEdm;

USE hakmanaEdm;

CREATE TABLE DeviceUser (                --User -> DeviceUser
    userNIC VARCHAR(11) PRIMARY KEY,
    name VARCHAR(50),
    title VARCHAR(25),
    gmail VARCHAR(35),
);

CREATE TABLE Desktop (
    DesRegNum VARCHAR(13) PRIMARY KEY,   -- RegNum -> DesRegNum(HAK/DES/0001)
    model VARCHAR(50), -- HP PRO TOWER 280 G9 PCI
    status VARCHAR(10), --active,inactive,repairing
    serialNum VARCHAR(20), --5CD6123LMN
    purchasedFrom VARCHAR(50), --Institute :TULIP COMPUTERS- MATARA
    ram VARCHAR(8),  --4 GB,8 GB,12 GB,...
    processor VARCHAR(50),  --AMD Ryzen 7 5800X,Intel(R) Core(TM) i7 Processors (14th gen)
    hardDisk VARCHAR(8),  --1 TB,500 GB
    os VARCHAR(10),       --WINDOWS,LINUX
    floppyDisk VARCHAR(3), --yes,no
    soundCard VARCHAR(15), --onboard,dedicated
    tvCard VARCHAR(15),        --yes/no
    networkCard VARCHAR(15),  --onboard,dedicated,no
    ssd VARCHAR(13),      --1 TB,500 GB To enter the value  Newly added
    cdRom VARCHAR(3),      --yes,no   (Newly added)
    monitorRegNum VARCHAR(13),  --HAK/MON/0001
    speakerRegNum VARCHAR(13),  --HAK/SPE/0001
    mouseRegNum VARCHAR(13),    --HAK/MOU/0001
    keyboardRegNum VARCHAR(13), --HAK/KEY/0001
    micRegNum VARCHAR(13),      --HAK/MIC/0001
	scannerRegNum VARCHAR(13),  --HAK/SCA/0001
	printerRegNum VARCHAR(13),  --HAK/PRI/0001  Newly added
	upsRegNum VARCHAR(13),      --HAK/UPS/0001  Newly added
	powerSupplyRegNum VARCHAR(13), --HAK/POW/0001  Newly added
    userNIC VARCHAR(15),
    FOREIGN KEY (userNIC) REFERENCES DeviceUser (userNIC)
     --Remove projectorRegNum,warranty
);

CREATE TABLE Printer (
    PrinterRegNum VARCHAR(13) PRIMARY KEY NOT NULL, --regNum -> PrinterRegNum
    model VARCHAR(50), --CANON IMAGE CLASS LBP 6030
    status VARCHAR(10), --active,inactive,repairing
    serialNum VARCHAR(30), --NJXA 369415
    purchasedFrom VARCHAR(50), --Institute :TULIP COMPUTERS- MATARA (Newly added)
    paperInput VARCHAR(15),  --150 SHEETS
    paperOutput VARCHAR(15), --150 SHEETS
    --Remove userNIC,warranty
);

CREATE TABLE Laptop (
    LaptopRegNum VARCHAR(13) PRIMARY KEY NOT NULL, --regNum -> LaptopRegNum  HAK/LAP/0001
    model VARCHAR(25), -- HP PRO TOWER 280 G9 PCI
    status VARCHAR(10), --active,inactive,repairing
    RAM VARCHAR(8),  --4 GB,8 GB,12 GB,...
    processor VARCHAR(50),  --AMD Ryzen 7 5800X,Intel(R) Core(TM) i7 Processors (14th gen)  CPU->processor
    hardDisk VARCHAR(8),  --1 TB,500 GB       storage -> hardDisk
    os VARCHAR(10),       --WINDOWS,LINUX     OperatingSystem -> os
    purchasedFrom VARCHAR(50), --Institute :TULIP COMPUTERS- MATARA (Newly added)
    mouseRegNum VARCHAR(13),    --HAK/MOU/0001
    keyboardRegNum VARCHAR(13), --HAK/KEY/0001
    userNIC VARCHAR(15),
    FOREIGN KEY (userNIC) REFERENCES DeviceUser (userNIC)
    --remove display,GraphicsCard
);

CREATE TABLE PhotoCopyMachine (
    PhotoCopyMachineRegNum VARCHAR(20) PRIMARY KEY NOT NULL, --regNum -> PhotoCopyMachineRegNum
    model VARCHAR(45), --Canon imageRUNNER ADVANCE C3520i
    status VARCHAR(10) --active,inactive,repairing
    purchasedFrom VARCHAR(50), --Institute :TULIP COMPUTERS- MATARA (Newly added)
    --Remove CopyingCapability
);

CREATE TABLE Monitor (                          --Monitors -> Monitor
    MonitorRegNum VARCHAR(20) PRIMARY KEY NOT NULL, --regNum ->  MonitorRegNum
    model VARCHAR(25), --HP
    screenSize VARCHAR(10), --19.5' (Newly Added)

    purchasedFrom VARCHAR(50), --Institute :TULIP COMPUTERS- MATARA (Newly added)

    --REMOVE userNIC,regNumDesktop
